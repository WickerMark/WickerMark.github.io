Setting up the Problem

import matplotlib.pyplot as plt
import numpy as np
import math
# Input parameters
angle = 30
u = 100
g = 9.81
h = 1000
A = 0.002
c_drag = 0.1
m = 0.01
dt = 0.01
density = 1.225
# Converts the angle to radians
theta = (angle * np.pi) / 180

# Resloves forces into horizontal and vertical component
costheta = math.cos(theta)
sintheta = math.sin(theta)

ux = u * costheta
uy = u * sintheta

No Air Resistance

# Calculates the range and time of flight of the projectile
R = (u**2 / g) * (sintheta*costheta+costheta*math.sqrt(sintheta**2 + ((2 * g * h) / (u**2))))
tflight = R / (u * costheta)

# Time of flight phrase
print("Time of flight: " + str(round(tflight, 2)) + " s")

# Range phrase
print("Range: " + str(round(R, 2)) + " m")   
Time of flight: 20.26 s
Range: 1754.37 m
# Creates an empty arrary for the x and y coordinates of the trajectory
x = []
y = []

# Plots the parabolic path of the projectile with a timestep of 0.01s across an interval of tflight
for i in np.arange(0, tflight, 0.01):
    t = i
    x.append(ux * t)
    T = t * t
    y.append(h + (uy * t) - (0.5 * g * T))
# Apogee calculation
apy = (0 - (uy*uy)) / (2 * g * -1) + h
y_graph = apy + apy * 0.1

Air Resistance

# Calculation for k
k = (0.5 * c_drag * density * A) / m
# Creates arrays for air resistance x and y values
x_plot = []
y_plot = []

# Sets initial x and y values
d_x = 0
d_y = h

# Adds initial x and y values to the arrays
x_plot.append(d_x)
y_plot.append(d_y)

# Verlet method to update the acceleration, velocity and position of the projectile after each discrete timestep
n = 0
t_drag = 0
while y_plot[n] >= 0:
    t_drag = t_drag + dt
    ax = - (ux / u) * k * u**2
    ay = - g - (uy / u) * k * u**2
    d_x = d_x + ux * dt + 0.5 * ax * dt**2
    d_y = d_y + uy * dt + 0.5 * ay * dt**2
    x_plot.append(d_x)
    y_plot.append(d_y)
    ux = ux + ax * dt
    uy = uy + ay * dt
    u = math.sqrt(ux**2 + uy**2)
    n += 1

R_drag = x_plot[n]
# Generates the calculated values text
# Time of flight phrase
print("Time of flight (drag): " + str(round(t_drag, 2)) + " s")
    
# Range phrase
print("Range (drag): " + str(round(R_drag, 2)) + " m")
Time of flight (drag): 41.15 s
Range (drag): 179.22 m

Air Resistance and Varying Air Density with Height

# Parameters for non linear air resistance
# Sea level standard atmospheric pressure
p0 = 101325
# Intial height
h0 = h
# Specific gas constant for air
sgc = 287.052874
# mass of one air molecule
mass_am = 4.81e-26
# Boltzmann constant
kB = 1.380649e-23
# Initial temperature in Celsius (standard temperature at sea level)
T0 = 15

x_air = 0
y_air = h

ux = u * costheta
uy = u * sintheta

d_x = 0
d_y = h
# Verlet method for varying air density
def verlet_air(p0, h0, sgc, mass_am, kB, T0, d_x, d_y, ux, uy, m, A, dt, c_drag, u, h, density):
    t = 0
    x_air_plot = []
    y_air_plot = []

    x_air_plot.append(d_x)
    y_air_plot.append(d_y)
    
    n = 0
    while y_air_plot[n] >= 0:
        k = (0.5 * c_drag * density * A) / m

        ax = - (ux / u) * k * u**2
        ay = - g - (uy / u) * k * u**2
        d_x = d_x + ux * dt + 0.5 * ax * dt**2
        d_y = d_y + uy * dt + 0.5 * ay * dt**2
        x_air_plot.append(d_x)
        y_air_plot.append(d_y)
        ux = ux + ax * dt
        uy = uy + ay * dt
        u = math.sqrt(ux**2 + uy**2)
        h = d_y

        temp = T0 - h * 0.00650
        temp = temp + 273.15
        p = p0 * math.exp((-mass_am * g * h) / (kB * temp))
        density = p / (sgc * temp)

        n += 1

        t = t + dt

    return x_air_plot, y_air_plot, n, t

x_air_plot, y_air_plot, n, t_air = verlet_air(p0, h0, sgc, mass_am, kB, T0, x_air, y_air, ux, uy, m, A, dt, c_drag, u, h, density)

R_air = x_air_plot[n]
# Generates the calculated values text
# Time of flight phrase
print("Time (varying air density): " + str(round(t_air, 2)) + " s")

# Range phrase
print("Range (varying air density): " + str(round(R_air, 2)) + " m")    
Time (varying air density): 38.2 s
Range (varying air density): 97.53 m
# Creates the figure
fig, ax = plt.subplots()

# Sets the parameters of the plot
ax.set_xlabel("Displacement in x / m")
ax.set_ylabel("Displacement in y / m")
ax.set_ylim(0, y_graph)
ax.set_xlim(0, R)

# Components to be plotted
ax.plot(x, y, label="No Air Resistance")
ax.grid()
ax.plot(x_plot, y_plot, label="Air Resistance")
ax.plot(x_air_plot, y_air_plot, label = "Air Resistance and Varying Air Density")

ax.legend()
